\begin{circuitikz} [scale=0.6, transform shape]
    \ctikzset{resistor = european}

    % --- OP1 ---
    \node[op amp](OP1) {$OP1$};
    
    \draw (OP1.+)
    -- ++(-0.3, 0)
    to[short, o-, l=$v$] ++(0, 0);

    \draw (OP1.-)
    -- ++(0, 1)
    coordinate(t1)
    -- (t1 -| OP1.out)
    to[short, -*] (OP1.out);

    % --- OP2 ---
    \draw (OP1.out)
    to[R=$R1$] ++(2, 0)
    node[op amp, anchor=-](OP2) {$OP2$};

    \draw (OP2.+)
    node[ground]{};

    \draw (OP2.-)
    to[short, *-] ++(0, 1)
    coordinate(t2)
    to[R=$R1$] (t2 -| OP2.out)
    to[short, -*] (OP2.out);

    % --- OP3 ---
    \draw (OP2.out)
    to[R=$R3$] ++(2, 0)
    node[op amp, anchor=-](OP3) {$OP3$};

    \draw (OP3.+)
    ++(-1, 0)
    to[short, o-, l=$+15V$] ++(0, 0)
    to[potentiometer, l_=$R3$, n=R3] ++(0, -2)
    to[short, -o, l=$-15V$] ++(0, 0);

    \draw (OP3.+)
    -- (OP3.+ |- R3.wiper)
    -- (R3.wiper);

    \draw (OP3.-)
    -- ++(0, 1)
    coordinate(t3)
    to[potentiometer, l=$R4$, n=R4] (t3 -| OP3.out)
    -- (OP3.out);

    \draw (t3)
    -- (t3 |- R4.wiper)
    -- (R4.wiper);

    \draw (OP3.out)
    to[short, *-o, l_=$\theta_m$] ++(0.3, 0);
    
\end{circuitikz}
